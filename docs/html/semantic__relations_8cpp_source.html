123std::vector<InformationState> generateSubStates(int n, int k) {
124 std::vector<InformationState> result;
125
126 if (k == 0) {
127 result.push_back(InformationState());
128 return result;
129 }
130
131 if (k > n + 1) {
132 return result;
133 }
134
135 int estimate = 1;
136 for (int i = 1; i <= k; i++) {
137 estimate = estimate * (n + 2 - i) / i;
138 }
139 result.reserve(estimate);
140
141 std::function<void(int, InformationState&)> backtrack =
142 [&](int start, InformationState& current) {
143 if (current.size() == k) {
144 result.push_back(current);
145 return;
146 }
147
148 ReferentSystem r;
149
150 for (int i = start; i <= n; ++i) {
151 Possibility p(std::make_shared<ReferentSystem>(r), i);
152 current.insert(p);
153
154 backtrack(i + 1, current);
155
156 current.erase(p);
157 }
158 };
159
160 InformationState current;
161 backtrack(0, current);
162
163 return result;
164}

183{
184 for (const int i : std::views::iota(0, model.world_cardinality())) {
185 std::vector<InformationState> states = generateSubStates(model.world_cardinality() - 1, i);

312std::expected<bool, std::string> similar(const Possibility& p1, const Possibility& p2)
313{
314 const auto have_same_denotation = [&](std::string_view variable) -> bool {
315 const auto denotation_at_p1 = variableDenotation(variable, p1);
316 const auto denotation_at_p2 = variableDenotation(variable, p2);
317 if (!denotation_at_p1.has_value()) {
318 throw std::out_of_range(denotation_at_p1.error());

320 if (!denotation_at_p2.has_value()) {

385std::expected<bool, std::string> equivalent(const QMLExpression::Expression& expr1, const QMLExpression::Expression& expr2, const IModel& model)
386{
