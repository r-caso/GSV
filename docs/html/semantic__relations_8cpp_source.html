103std::vector<InformationState> generateSubStates(int n, int k) {
104 std::vector<InformationState> result;
105
106 if (k == 0) {
107 result.push_back(InformationState());
108 return result;
109 }
110
111 if (k > n + 1) {
112 return result;
113 }
114
115 int estimate = 1;
116 for (int i = 1; i <= k; i++) {
117 estimate = estimate * (n + 2 - i) / i;
118 }
119 result.reserve(estimate);
120
121 std::function<void(int, InformationState&)> backtrack =
122 [&](int start, InformationState& current) {
123 if (current.size() == k) {
124 result.push_back(current);
125 return;
126 }
127
128 ReferentSystem r;
129
130 for (int i = start; i <= n; ++i) {
131 Possibility p(std::make_shared<ReferentSystem>(r), i);
132 current.insert(p);
133
134 backtrack(i + 1, current);
135
136 current.erase(p);
137 }
138 };
139
140 InformationState current;
141 backtrack(0, current);
142
143 return result;
144}

209bool entails(const std::vector<Expression>& premises, const Expression& conclusion, const IModel& model)
210{
211 for (const int i : std::views::iota(0, model.world_cardinality())) {
212 std::vector<InformationState> states = generateSubStates(model.world_cardinality() - 1, i);
213 for (InformationState& input_state : states) {
214 try {
