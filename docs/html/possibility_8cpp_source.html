60bool extends(const Possibility& p2, const Possibility& p1)
61{
62 const auto peg_is_new_or_maintains_assignment = [&](const std::pair<int, int>& map) -> bool {
63 const int peg = map.first;
64
65 return !p1.assignment.contains(peg) || (p1.assignment.at(peg) == p2.assignment.at(peg));
66 };
67
68 return (p1.world == p2.world) && std::ranges::all_of(p2.assignment, peg_is_new_or_maintains_assignment);
69}

90std::expected<int, std::string> variableDenotation(std::string_view variable, const Possibility& p)
91{
92 const auto peg = p.referentSystem->value(variable);
93
94 if (!peg.has_value()) {
95 return std::unexpected(peg.error());
96 }
97
98 // Whenever variable exists in referent system, assignment is guaranteed to
99 // contain the corresponding peg, so there is no need to check for existence
100 // before returning
101 return p.assignment.at(peg.value());
102}
