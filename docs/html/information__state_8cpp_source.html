109InformationState update(const InformationState& input_state, std::string_view variable, int individual)
110{
111 InformationState output_state(input_state.model, false);
112
113 auto r_star = std::make_shared<ReferentSystem>();
114
115 for (const auto& p : input_state.possibilities) {
116 Possibility p_star(r_star, p.world);
117 p_star.assignment = p.assignment;
118 r_star->pegs = p.referentSystem->pegs;
119 for (const auto& map : p.referentSystem->variablePegAssociation) {
120 auto var = map.first;
121 int peg = map.second;
122 r_star->variablePegAssociation[var] = peg;
123 }
124
125 p_star.update(variable, individual);
126
127 output_state.possibilities.insert(p_star);
128 }
129
130 return output_state;
131}

142bool extends(const InformationState& s2, const InformationState& s1)
143{
