43InformationState Evaluator::operator()(const std::shared_ptr<UnaryNode>& expr, std::variant<std::pair<InformationState, const IModel*>> params) const
44{
45 InformationState hypothetical_update = std::visit(Evaluator(), expr->scope, params);
46 InformationState& input_state = (std::get<std::pair<InformationState, const IModel*>>(params)).first;
47
48 if (expr->op == Operator::E_POS) {
49 if (hypothetical_update.empty()) {
50 input_state.clear();
51 }
52 }
53 else if (expr->op == Operator::E_NEC) {
54 if (!subsistsIn(input_state, hypothetical_update)) {
55 input_state.clear();
56 }
57 }
58 else if (expr->op == Operator::NEG) {
59 filter(input_state, [&](const Possibility& p) -> bool { return !subsistsIn(p, hypothetical_update); });
60 }
61 else {
62 throw(std::invalid_argument("Invalid operator for unary formula"));
63 }
64
65 return std::move(input_state);
66}

145InformationState Evaluator::operator()(const std::shared_ptr<QuantificationNode>& expr, std::variant<std::pair<InformationState, const IModel*>> params) const
146{

148 const IModel* model = (std::get<std::pair<InformationState, const IModel*>>(params)).second;
149
150 if (expr->quantifier == Quantifier::EXISTENTIAL) {
151 std::vector<InformationState> all_state_variants;
152
153 for (const int i : std::views::iota(0, model->domain_cardinality())) {

155 all_state_variants.push_back(std::visit(
156 Evaluator(),
157 expr->scope,
158 std::variant<std::pair<InformationState, const IModel*>>(std::make_pair(s_variant, model)))
159 );
160 }
161
162 InformationState output;
163 for (const auto& state_variant : all_state_variants) {
164 for (const auto& p : state_variant) {
165 output.insert(p);
166 }

170 }
171

176 const InformationState hypothetical_update = std::visit(
177 Evaluator(),
178 expr->scope,
179 std::variant<std::pair<InformationState, const IModel*>>(std::make_pair(update(input_state, expr->variable, d), model))
180 );
181 all_hypothetical_updates.push_back(hypothetical_update);
182 }
183
184 const auto subsists_in_all_hyp_updates = [&](const Possibility& p) -> bool {
185 const auto p_subsists_in_hyp_update = [&](const InformationState& hypothetical_update) -> bool {
186 return subsistsIn(p, hypothetical_update);
187 };
188 return std::ranges::all_of(all_hypothetical_updates, p_subsists_in_hyp_update);
189 };

192 }
193 else {
