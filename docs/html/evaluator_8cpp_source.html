74{
75 if (expr->op == Operator::CON) {

80 InformationState hypothetical_lhs = std::visit(Evaluator(), expr->lhs, state);
81

86 return hypothetical_lhs.contains(p) || hypothetical_rhs.contains(p);
87 };
88
89 filter(s, in_lhs_or_in_rhs);
90 }

92 InformationState hypothetical_consequent = std::visit(Evaluator(), expr->rhs, std::variant<InformationState>(hypothetical_lhs));
93

95 auto not_descendant_or_subsists = [&](const Possibility& p_star) -> bool {
96 return !isDescendantOf(p_star, p, hypothetical_lhs) || subsistsIn(p_star, hypothetical_consequent);
97 };
98 return std::ranges::all_of(hypothetical_lhs.possibilities, not_descendant_or_subsists);
99 };

102 return !subsistsIn(p, hypothetical_lhs) || all_descendants_subsist(p);
103 };
104

106 }

125InformationState Evaluator::operator()(std::shared_ptr<QuantificationNode> expr, std::variant<InformationState> state) const
126{
127 InformationState& s = std::get<InformationState>(state);
128
129 if (expr->quantifier == Quantifier::EXISTENTIAL) {
130 std::vector<InformationState> all_state_variants;
131
132 for (int i : std::views::iota(0, s.model.domain_cardinality())) {
133 InformationState s_variant = update(s, expr->variable, i);
134 all_state_variants.push_back(std::visit(Evaluator(), expr->scope, std::variant<InformationState>(s_variant)));
135 }
136
137 InformationState output(s.model, false);
138 for (const auto& state_variant : all_state_variants) {
139 for (const auto& p : state_variant.possibilities) {
140 output.possibilities.insert(p);
141 }
142 }
143
144 return output;
145 }
146 else if (expr->quantifier == Quantifier::UNIVERSAL) {
147 std::vector<InformationState> all_hypothetical_updates;
148
149 for (int d : std::views::iota(0, s.model.domain_cardinality())) {
150 InformationState hypothetical = std::visit(Evaluator(), expr->scope, std::variant<InformationState>(update(s, expr->variable, d)));
151 all_hypothetical_updates.push_back(hypothetical);
152 }
153
154 const auto subsists_in_all_hyp_updates = [&](const Possibility& p) -> bool {
155 const auto p_subsists_in_hyp_update = [&](const InformationState& hypothetical) -> bool {
156 return subsistsIn(p, hypothetical);
157 };
158 return std::ranges::all_of(all_hypothetical_updates, p_subsists_in_hyp_update);
159 };

162 }
163 else {
164 throw(std::invalid_argument("Invalid quantifier"));
