53InformationState Evaluator::operator()(std::shared_ptr<UnaryNode> expr, std::variant<std::pair<InformationState, const IModel*>> params) const
54{
55 InformationState hypothetical_update = std::visit(Evaluator(), expr->scope, params);
56 InformationState& input_state = (std::get<std::pair<InformationState, const IModel*>>(params)).first;
57
58 if (expr->op == Operator::E_POS) {
59 if (hypothetical_update.empty()) {
60 input_state.clear();
61 }
62 }
63 else if (expr->op == Operator::E_NEC) {
64 if (!subsistsIn(input_state, hypothetical_update)) {
65 input_state.clear();
66 }
67 }
68 else if (expr->op == Operator::NEG) {
69 filter(input_state, [&](const Possibility& p) -> bool { return !subsistsIn(p, hypothetical_update); });
70 }
71 else {
72 throw(std::invalid_argument("Invalid operator for unary formula"));
73 }
74
75 return std::move(input_state);
76}

155InformationState Evaluator::operator()(std::shared_ptr<QuantificationNode> expr, std::variant<std::pair<InformationState, const IModel*>> params) const
156{
157 InformationState& input_state = (std::get<std::pair<InformationState, const IModel*>>(params)).first;
158 const IModel* model = (std::get<std::pair<InformationState, const IModel*>>(params)).second;
159
160 if (expr->quantifier == Quantifier::EXISTENTIAL) {
161 std::vector<InformationState> all_state_variants;
162
163 for (int i : std::views::iota(0, model->domain_cardinality())) {
164 InformationState s_variant = update(input_state, expr->variable, i);
165 all_state_variants.push_back(std::visit(
166 Evaluator(),
167 expr->scope,
168 std::variant<std::pair<InformationState, const IModel*>>(std::make_pair(s_variant, model)))
169 );
170 }
171
172 InformationState output;
173 for (const auto& state_variant : all_state_variants) {
174 for (const auto& p : state_variant) {
175 output.insert(p);
176 }
177 }
178
179 return output;
180 }
181 else if (expr->quantifier == Quantifier::UNIVERSAL) {
182 std::vector<InformationState> all_hypothetical_updates;
183
184 for (int d : std::views::iota(0, model->domain_cardinality())) {
185 InformationState hypothetical_update = std::visit(
186 Evaluator(),
187 expr->scope,
188 std::variant<std::pair<InformationState, const IModel*>>(std::make_pair(update(input_state, expr->variable, d), model))
189 );
190 all_hypothetical_updates.push_back(hypothetical_update);
191 }
192
193 const auto subsists_in_all_hyp_updates = [&](const Possibility& p) -> bool {
194 const auto p_subsists_in_hyp_update = [&](const InformationState& hypothetical_update) -> bool {
195 return subsistsIn(p, hypothetical_update);
196 };
197 return std::ranges::all_of(all_hypothetical_updates, p_subsists_in_hyp_update);
198 };
199
200 filter(input_state, subsists_in_all_hyp_updates);
201 }
202 else {
203 throw(std::invalid_argument("Invalid quantifier"));
204 }
205
206 return std::move(input_state);
207}
